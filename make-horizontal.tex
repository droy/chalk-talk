%% Note: the correct use of this file requires that you specify below the number of pages in the presentation and the the page dimensions.  I've not tested more than 26 slides.  I suspect that the approach may break the pdfpages package or Abobe Acrobat at some point.

%% To view the wide slides produced, use a program like Adobe Reader.  In Full Screen mode, press Ctrl+Y to change the zoom level, and choose "Fit Height".

\documentclass[reqno,oneside,onecolumn,8pt]{amsart}
\usepackage[utf8]{inputenc} 
\usepackage{pdfpages,calc}

\newcounter{numpages}
\setcounter{numpages}{6} % set the number of pages...
\newcounter{pagewidth}
\setcounter{pagewidth}{80} % and their width in 'mm'
\newcounter{totalwidth}
\setcounter{totalwidth}{\thenumpages*\thepagewidth}

\usepackage[
	paperwidth={\thetotalwidth}mm, % 24 x 64mm,
	paperheight=120mm,
	top=3.5mm,%
	bottom=3.5mm,%
	left=1.75mm,%
	right=1.75mm,%
	headsep=6.5mm,%
	footskip=8.5mm%
	]{geometry}
	
\begin{document}
\includepdf[nup={\thenumpages}x1,pages={1-\thenumpages},frame=true]{slides.pdf}
\end{document}
